% define variable for localized "of" for "Page X of Y"
\newcommand{\pageof}{
    \iflanguageloaded{german}{Seite \thepage~von \letterlastpage}{}
    \iflanguageloaded{english}{Page \thepage~of \letterlastpage}{}
}

    \usekomafont{pagenumber}{\pageof}
